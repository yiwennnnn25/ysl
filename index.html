📋 Supabase SQL — run this once in your project 
create table purchases (
  id uuid default gen_random_uuid() primary key,
  item_name text,
  category text,
  brand text,
  price_myr numeric,
  score numeric,
  decision text,
  occasions text[],
  purchased_at timestamptz default now(),
  feedback_quality integer,
  feedback_sizing text,
  feedback_notes text,
  feedback_thumb text,
  feedback_at timestamptz
);

create table brands (
  id uuid default gen_random_uuid() primary key,
  name text unique,
  tier text,
  best_for text,
  notes text,
  thumb text,
  feedback_count integer default 0,
  created_at timestamptz default now()
);

-- NEW: persists every evaluation so they can be revisited across devices.
create table evaluations (
  id text primary key,
  evaluated_at timestamptz default now(),
  item_name text,
  category text,
  price numeric,
  currency text,
  price_myr numeric,
  occ text[],
  notes text,
  result jsonb,
  thumbs jsonb,
  chat jsonb,
  bought boolean default false,
  purchase_id uuid
);

-- IMPORTANT: the anon key is public — anyone with your project URL
-- + anon key can read/write these tables. Either keep both strictly
-- private, or enable Supabase Auth and lock down with RLS:
--   alter table purchases enable row level security;
--   alter table brands enable row level security;
--   alter table evaluations enable row level security;
--   (then add policies tied to auth.uid())
-- Note: RLS with a permissive `using (true)` policy gives no real
-- protection — it needs actual auth to be meaningful.
